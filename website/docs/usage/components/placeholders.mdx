    <img alt="..." class="card-img-top" src="/twbs-helper-module/img/docs/image-cap.svg">

        <h5 class="card-title">Card title</h5>

            <span class="col-6 placeholder"></span>
        </div>
        <p class="card-text placeholder-glow">
            <span class="col-7 placeholder"></span>
            <span class="col-4 placeholder"></span>
            <span class="col-4 placeholder"></span>
            <span class="col-6 placeholder"></span>
            <span class="col-8 placeholder"></span>
        </p>
        <a aria-disabled="true" class="btn btn-primary col-6 disabled placeholder" href="#" role="button" tabindex="-1"></a>
    </div>
</div>` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

echo $this->card(
    [
        'image_top' => ['/twbs-helper-module/img/docs/image-cap.svg', ['alt' => '...',]],
        'title' => 'Card title',
        'text' => 'Some quick example text to build on the card title ' .
        'and make up the bulk of the card\'s content.',
        '<a class="btn btn-primary" href="&#x23;">Go somewhere</a>',
    ],
);

echo $this->card(
    [
        'image_top' => ['/twbs-helper-module/img/docs/image-cap.svg', ['alt' => '...',]],
        'title' => [
            'attributes' => [
                'class' => 'h5 placeholder-glow',
            ],
            'tag' => 'div',
            'content' => $this->placeholder(6),
        ],
        'text' => [
            'content' => $this->placeholder(7) .
            $this->placeholder(4) .
            $this->placeholder(4) .
            $this->placeholder(6) .
            $this->placeholder(8),
            'attributes' => ['class' => 'placeholder-glow'],
        ],
        $this->placeholder(['column' => 6, 'button' => ['options' => ['variant' => 'primary']]]),
    ],
    ['aria-hidden' => 'true']
);
```

</TabItem>
</Tabs>
