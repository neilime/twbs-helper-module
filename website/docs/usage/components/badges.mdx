<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<button class="btn btn-primary position-relative" name="button" type="button" value="">

        <span class="visually-hidden">unread messages</span>
    </span>

<button class="btn btn-primary position-relative" name="button" type="button" value="">

  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

echo $this->formButton()->renderSpec([
    'options' => [
        'label' => 'Inbox',
        'badge' => [
            '99+',
            [
                'variant' => 'danger',
                'positioned' => true,
                'type' => 'pill',
                'hidden_content' => 'unread messages',
            ],
        ],
        'variant' => 'primary',
    ],
]);

echo PHP_EOL . '<br />';

echo $this->formButton()->renderSpec([
    'options' => [
        'label' => 'Profile',
        'badge' => [
            '',
            [
                'variant' => 'danger',
                'positioned' => true,
                'type' => 'pill',
                'hidden_content' => 'New alerts',
            ],
        ],
        'variant' => 'primary',
    ],
]);
```
