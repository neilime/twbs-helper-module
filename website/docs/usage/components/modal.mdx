```php
<?php

echo $this->modal(
    [
        'title' => 'Modal title',
        'Modal body text goes here.',
        'footer' => [
            'button' => [
                [
                    'options'    => [
                        'label'   => 'Close',
                        'variant' => 'secondary',
                    ],
                    'attributes' => [    'data-dismiss' => 'modal'    ],
                ],
                [
                    'options' => [
                        'label'   => 'Save changes',
                        'variant' => 'primary',
                    ],
                ],
            ],
        ]
    ]
);
```
