    'attributes' => ['href' => '#'],

    'options' => [
        'label' => 'Button',
        'variant' => 'primary',
    ],
    'attributes' => ['type' => 'submit'],

echo $this->formButton()->renderSpec([
    'options' => [
        'tag' => 'input',

        'variant' => 'primary',
    ],
    'attributes' => ['type' => 'button'],

echo $this->formButton()->renderSpec([

        'tag' => 'input',
