```php
<?php


echo $this->breadcrumbs(
    new \Laminas\Navigation\Navigation(
        [
            [
                'label'  => 'Home',
                'uri'    => '/',
                'active' => true,
            ],
        ]
    )
)->setMinDepth(0);

echo $this->breadcrumbs(
    new \Laminas\Navigation\Navigation(
        [
            [
                'label' => 'Home',
                'uri'   => '/',
                'pages' => [
                    [
                        'label'  => 'Library',
                        'uri'    => '/library',
                        'active' => true,
                    ],
                ],
            ],
        ]
    )
)->setMinDepth(0);

echo $this->breadcrumbs(
    new \Laminas\Navigation\Navigation(
        [
            [
                'label' => 'Home',
                'uri'   => '/',
                'pages' => [
                    [
                        'label' => 'Library',
                        'uri'   => '/library',
                        'pages' => [
                            [
                                'label'  => 'Data',
                                'uri'    => '/library/data',
                                'active' => true,
                            ],
                        ],
                    ],
                ],
            ],
        ]
    )
)->setMinDepth(0);
```
