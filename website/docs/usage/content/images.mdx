### Float

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<img alt="..." class="float-start rounded" src="/twbs-helper-module/img/docs/200x200.svg">

  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

    '/twbs-helper-module/img/docs/200x200.svg',
    [
        'rounded' => true,
        'alt'     => '...',
        'class'   => 'float-start',
    ]
);

echo PHP_EOL;

echo $this->image(
    '/twbs-helper-module/img/docs/200x200.svg',
    [
        'rounded' => true,
        'alt'     => '...',
        'class'   => 'float-end',
    ]
);

```

</TabItem>
