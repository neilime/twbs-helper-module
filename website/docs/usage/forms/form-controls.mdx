    </div>
    <div class="mb-3">

<TabItem value="source" label="Source">

```php
<?php

$factory = new \Laminas\Form\Factory();

echo $this->form($factory->create([
    'type' => 'form',
    'elements' => [
        [
            'spec' => [
                'name' => 'email',
                'options' => [
                    'label' => 'Email address'
                ],
                'attributes' => [
                    'type' => 'email',
                    'id' => 'exampleFormControlInput1',
                    'placeholder' => 'name@example.com',
                ],
            ],
        ],
        [
            'spec' => [
                'name' => 'textarea',
                'options' => [
                    'label' => 'Example textarea'
                ],
                'attributes' => [
                    'type' => 'textarea',
                    'id' => 'exampleFormControlTextarea1',
                    'rows' => 3,
                ],
            ],
        ],
    ],
]));
```

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<input aria-label=".form-control-lg example" class="form-control form-control-lg" name="lg" placeholder=".form-control-lg" type="text" value="">
<br>
<input aria-label="default input example" class="form-control" name="default" placeholder="Default input" type="text" value="">
<br>
<input aria-label=".form-control-sm example" class="form-control form-control-sm" name="sm" placeholder=".form-control-sm" type="text" value="">` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

$factory = new \Laminas\Form\Factory();

// Render large input
$element = $factory->create([
    'name' => 'lg',
    'type' => 'text',
    'options' => ['size' => 'lg'],
    'attributes' => ['placeholder' => '.form-control-lg', 'aria-label' => '.form-control-lg example'],
]);
echo $this->formElement($element);

echo PHP_EOL . '<br/>';

// Render default input
$element = $factory->create([
    'name' => 'default',
    'type' => 'text',
    'attributes' => ['placeholder' => 'Default input', 'aria-label' => 'default input example'],
]);
echo $this->formElement($element);

echo PHP_EOL . '<br/>';

// Render small input
$element = $factory->create([
    'name' => 'sm',
    'type' => 'text',
    'options' => ['size' => 'sm'],
    'attributes' => ['placeholder' => '.form-control-sm', 'aria-label' => '.form-control-sm example'],
]);
echo $this->formElement($element);
```

## Disabled

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.1/forms/form-control/#disabled)

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<input aria-label="Disabled input example" class="form-control" disabled name="disabled-input" placeholder="Disabled input" type="text" value="">
<br>
<input aria-label="Disabled input example" class="form-control" disabled name="disabled-readonly-input" readonly type="text" value="Disabled readonly input">` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

// Render disabled input
$factory = new \Laminas\Form\Factory();
$element = $factory->create([
    'name' => 'disabled-input',
    'type' => 'text',
    'attributes' => [
        'disabled' => true,
        'placeholder' => 'Disabled input',
        'aria-label' => 'Disabled input example'
    ],
]);
echo $this->formElement($element);

echo PHP_EOL . '<br/>';

// Render disabled and readonly input
$element = $factory->create([
    'name' => 'disabled-readonly-input',
    'type' => 'text',
    'attributes' => [
        'disabled' => true,
        'readonly' => true,
        'value' => 'Disabled readonly input',
        'aria-label' => 'Disabled input example'
    ],
]);
echo $this->formElement($element);
```

</Tabs>

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.1/forms/#readonly-plain-text)

```php
<?php

$factory = new \Laminas\Form\Factory();

// Render horizontal form
$form = $factory->create([
    'type' => 'form',
    'options' => ['layout' => \TwbsHelper\Form\View\Helper\Form::LAYOUT_HORIZONTAL],
    'elements' => [
        [
            'spec' => [
                'name' => 'email',
                'options' => [
                    'plaintext' => true,
                    'column' => 'sm-10',
                    'label' => 'Email',
                ],
                'attributes' => [
                    'type' => 'email',
                    'id' => 'staticEmail',
                    'value' => 'email@example.com',
                    'readonly' => true,
                ],
            ],
        ],
        [
            'spec' => [
                'name' => 'password',
                'options' => [
                    'column' => 'sm-10',
                    'label' => 'Password',
                ],
                'attributes' => [
                    'type' => 'password',
                    'id' => 'inputPassword',
                ],
            ],
        ],
    ],
]);

echo $this->form($form);

echo PHP_EOL . '<br/>';

// Render inline form
echo $this->form($factory->create([
    'type' => 'form',
    'options' => ['layout' => \TwbsHelper\Form\View\Helper\Form::LAYOUT_INLINE],
    'elements' => [
        [
            'spec' => [
                'name' => 'email',
                'options' => [
                    'plaintext' => true,
                    'label' => 'Email',
                    'show_label' => false,
                ],
                'attributes' => [
                    'type' => 'email',
                    'id' => 'staticEmail2',
                    'value' => 'email@example.com',
                    'readonly' => true,
                ],
            ],
        ],
        [
            'spec' => [
```
