## Default

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.3/forms/select/#default)

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.3">
      <div dangerouslySetInnerHTML={{ __html: `<select aria-label="Default select example" class="form-select" name="default-select">

    <option value="1">One</option>
    <option value="2">Two</option>
    <option value="3">Three</option>
</select>` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

$factory = new \Laminas\Form\Factory();

echo $this->formElement($factory->create([
    'name' => 'default-select',
    'type' => 'select',
    'options' => [
        'empty_option' => 'Open this select menu',
        'value_options' => [
            1 => 'One',
            2 => 'Two',
            3 => 'Three',
        ],
    ],
    'attributes' => [
        'aria-label' => 'Default select example',
    ],
]));
```

## Sizing

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.3/forms/select/#sizing)

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.3">
      <div dangerouslySetInnerHTML={{ __html: `<select aria-label=".form-select-lg example" class="form-select form-select-lg" name="lg-select">

    <option value="3">Three</option>
</select>
<br>
<select aria-label="size 3 select example" class="form-select" name="size-3-select" size="3">
    <option value="">Open this select menu</option>

    <option value="3">Three</option>

  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

$factory = new \Laminas\Form\Factory();

echo $this->formElement($factory->create([
    'name' => 'lg-select',
    'type' => 'select',
    'options' => [
        'size' => 'lg',
        'empty_option' => 'Open this select menu',
        'value_options' => [
            1 => 'One',
            2 => 'Two',
            3 => 'Three',
        ],
    ],
    'attributes' => [
        'aria-label' => '.form-select-lg example',
    ],
]));

echo PHP_EOL . '<br/>';

echo $this->formElement($factory->create([
    'name' => 'sm-select',
    'type' => 'select',
    'options' => [
        'size' => 'sm',
        'empty_option' => 'Open this select menu',
        'value_options' => [
            1 => 'One',
            2 => 'Two',
            3 => 'Three',
        ],
    ],
    'attributes' => [
        'aria-label' => '.form-select-sm example',
    ],
]));

echo PHP_EOL . '<br/>';

echo $this->formElement($factory->create([
    'name' => 'multiple-select',
    'type' => 'select',
    'options' => [
        'empty_option' => 'Open this select menu',
        'value_options' => [
            1 => 'One',
            2 => 'Two',
            3 => 'Three',
        ],
    ],
    'attributes' => [
        'multiple' => true,
        'aria-label' => 'multiple select example',
    ],
]));

echo PHP_EOL . '<br/>';

echo $this->formElement($factory->create([
    'name' => 'size-3-select',
    'type' => 'select',
    'options' => [
        'empty_option' => 'Open this select menu',
        'value_options' => [
            1 => 'One',
            2 => 'Two',
            3 => 'Three',
        ],
    ],
    'attributes' => [
        'size' => 3,
        'aria-label' => 'size 3 select example',
    ],
]));
```
