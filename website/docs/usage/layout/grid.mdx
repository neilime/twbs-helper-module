        [
            ['Level 1: .col-sm-3', ['column' => 'sm-3']],
            [
                null,
                [
                    'column' => 'sm-9',
                    'grid' => [
                        'rows' => [
                            [
                                [
                                    ['Level 2: .col-8 .col-sm-6', ['column' => [8, 'sm-6']]],
                                    ['Level 2: .col-4 .col-sm-6', ['column' => [4, 'sm-6']]],
